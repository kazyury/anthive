                def root_directory=(dir)
                        raise AntHiveError.new("Root-Directory [ #{dir} ] does not exist.") unless File.directory?(dir)
                        @root_directory=dir
                        @cache_file=".#{@root_directory.gsub(/\n/,'.')}.cache"
                        @logger.debug(self.class){"@root_directory was set to [ #{dir} ]."}
                end
