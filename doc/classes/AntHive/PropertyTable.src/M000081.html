                        process_1=false
                        process_2=false
                        buffer=""
                        result=[]

                        str.split(//).each do |s|
                                case s
                                when '$'
                                        process_1=true
                                when '{'
                                        if process_1==true
                                                process_1=false
                                                process_2=true
                                                buffer=""
                                        end
                                when '}'
                                        process_2=false
                                        result.push buffer unless buffer==""
                                        buffer=""
                                else
                                        process_1=false if process_1==true
                                        buffer << s if process_2==true
                                end
                        end

                        result
                end
